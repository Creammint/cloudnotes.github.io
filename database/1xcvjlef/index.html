列转行

SELECT * FROM (
    SELECT snd,cls,sum(qty) qty
    FROM sdrpts 
    where cls in ('零售','零售退') 
    and snd = '1000560' 
    --and gdgid = '3027401'
    and fildate >= to_date('20231201','yyyyMMdd')
    and fildate < to_date('20231202','yyyyMMdd')
    group by snd,cls
) PIVOT(max(qty) FOR cls IN ('零售' as lsTotal, '零售退' as lstToal));

求占百分比

select deptno , sum(sal) ,
       round(ratio_to_report( sum(sal) ) over() , 5)*100 || '%' percent_sum 
from emp
group by deptno ;

select deptno , --sum(sal) ,
round(ratio_to_report( sal ) over(partition by deptno) , 5)*100 || '%' percent_deptno 
from emp
order by deptno , sal ;
